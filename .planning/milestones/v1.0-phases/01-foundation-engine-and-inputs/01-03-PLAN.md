### NumberSliderInput.vue — reusable coupled number + slider (per CONTEXT.md Input Interaction Design)

```vue
<script setup lang="ts">
const props = defineProps<{
  modelValue: number
  label: string
  min: number
  max: number
  step?: number
  unit?: string
}>()
const emit = defineEmits<{ 'update:modelValue': [value: number] }>()

function onInput(e: Event) {
  const val = Number((e.target as HTMLInputElement).value)
  if (!isNaN(val)) emit('update:modelValue', val)
}
</script>

<template>
  <div class="flex flex-col gap-1">
    <label class="text-sm font-medium text-gray-700">
      {{ label }}
      <span v-if="unit" class="text-gray-400 font-normal ml-1">{{ unit }}</span>
    </label>
    <div class="flex items-center gap-3">
      <input
        type="number"
        :value="modelValue"
        :min="min"
        :max="max"
        :step="step ?? 1"
        class="w-24 px-2 py-1 border border-gray-300 rounded text-sm text-right
               focus:outline-none focus:ring-2 focus:ring-blue-500"
        @input="onInput"
        @change="onInput"
      />
      <input
        type="range"
        :value="modelValue"
        :min="min"
        :max="max"
        :step="step ?? 1"
        class="flex-1 accent-blue-600"
        @input="onInput"
      />
    </div>
  </div>
</template>
```

Number input is the primary control (architects enter exact values). Slider is coupled helper for quick sensitivity exploration. Both update on input event — no debounce needed for Pinia `ref()` writes.

### HostSpecsForm.vue — cores, sockets, RAM, storage, host count (HOST-01 through HOST-05)
Includes VCFA core blocker warning (MGMT-07, HOST-06).

```vue
<script setup lang="ts">
import { computed } from 'vue'
import { useI18n } from 'vue-i18n'
import { useInputStore } from '@/stores/inputStore'
import { useCalculationStore } from '@/stores/calculationStore'
import { storeToRefs } from 'pinia'
import NumberSliderInput from '@/components/shared/NumberSliderInput.vue'
import WarningBanner from '@/components/shared/WarningBanner.vue'
const { t } = useI18n()
const input = useInputStore()
const calc = useCalculationStore()
const { coresPerSocket, socketsPerHost, hostRamGB, hostStorageTB, hostCount } = storeToRefs(input)
const { validationErrors } = storeToRefs(calc)

const vcfaBlockerError = computed(() =>
  validationErrors.value.find(e => e.code === 'VCFA_MIN_CORES' && e.severity === 'error')
)
const totalCoresPerHost = computed(() => coresPerSocket.value * socketsPerHost.value)
</script>

<template>
  <section class="bg-white rounded-lg border border-gray-200 p-4 space-y-4">
    <h2 class="text-base font-semibold text-gray-900">{{ t('host.label') }}</h2>

    <!-- VCFA blocker — must be unmissable (MGMT-07) -->

      v-if="vcfaBlockerError"
      :message="t('warnings.vcfaMinCores')"
      severity="error"
    />

    <div class="grid grid-cols-1 sm:grid-cols-2 gap-4">
      <NumberSliderInput
        v-model="coresPerSocket"
        :label="t('host.coresPerSocket')"
        :min="4" :max="64" :step="2"
      />
      <NumberSliderInput
        v-model="socketsPerHost"
        :label="t('host.socketsPerHost')"
        :min="1" :max="8" :step="1"

        {{ t('host.totalCores') }}:
        <span
          :class="totalCoresPerHost < 12 ? 'text-red-600 font-bold' : 'text-gray-900 font-semibold'"
        >
          {{ totalCoresPerHost }}

      </div>

        unit="GB"
```
